%% improving tuftebreak
%% http://tex.stackexchange.com/questions/291746/tufte-latex-newthought-after-section
\makeatletter
\def\tuftebreak{%
	\if@nobreak\else
	\par
	\ifdim\lastskip<\tufteskipamount
	\removelastskip \penalty -100
	\tufteskip
	\fi
	\fi
}
\makeatother
